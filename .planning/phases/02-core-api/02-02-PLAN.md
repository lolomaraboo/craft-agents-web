```typescript
import fp from 'fastify-plugin'
import { FastifyPluginAsync } from 'fastify'
import { sessionsRoutes } from './sessions.js'
import { workspacesRoutes } from './workspaces.js'
import { configRoutes } from './config.js'
import { credentialsRoutes } from './credentials.js'
import { mcpRoutes } from './mcp.js'
import { themeRoutes } from './theme.js'
import { SessionManager } from '../../lib/session-manager.js'

// Type augmentation for fastify decorators
declare module 'fastify' {
  interface FastifyInstance {
    sessionManager: SessionManager
  }
}

const apiRoutesPlugin: FastifyPluginAsync = async (fastify) => {
  // Decorate with SessionManager instance (already from 02-01)
  const sessionManager = new SessionManager()
  fastify.decorate('sessionManager', sessionManager)

  // Register all API routes under /api prefix
  await fastify.register(sessionsRoutes, { prefix: '/api' })      // from 02-01
  await fastify.register(workspacesRoutes, { prefix: '/api' })    // NEW
  await fastify.register(configRoutes, { prefix: '/api' })        // NEW
  await fastify.register(credentialsRoutes, { prefix: '/api' })   // NEW (API-06)
  await fastify.register(mcpRoutes, { prefix: '/api' })           // NEW
  await fastify.register(themeRoutes, { prefix: '/api' })         // NEW
}

export const apiRoutes = fp(apiRoutesPlugin, {
  name: 'api-routes',
  fastify: '5.x'
})
```
